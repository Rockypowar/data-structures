int search(int ary[], int size, int key) {
int mid, loc = -1, lb = 0, up = size - 1;
while (lb <= up)
{ // Change the condition to lb <= up
mid = (lb + up) / 2;
if (ary[mid] == key)
{ loc = mid;
break;
} else if (key < ary[mid])
{ up = mid - 1;
} else { lb = mid + 1;
} } return loc;
} int bubble(int ary[], int size)
{ int i, j, temp;
for (i = size - 1; i > 0; i--)
{ for (j = 0; j <= i; j++)
{ if (ary[j] > ary[j + 1])
{ temp = ary[j];
ary[j] = ary[j + 1];
ary[j + 1] = temp; }
} } return 0;
} void main()
{ int ary[10], i, count, loc, num;
printf("Enter your array size : ");
scanf("%d", &count);
for (i = 0; i < count; i++)
{ printf("\nEnter %d value : ", i + 1);
scanf("%d", &ary[i]); }
printf("\nEnter the number you want to search : ");
scanf("%d", &num);
bubble(ary, count);
loc = search(ary, count, num);
if (loc == -1)
{ printf("\nElement not found in array.");
} else
{ printf("Element found at position : %d", loc + 1);
} }
