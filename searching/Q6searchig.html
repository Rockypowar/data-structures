void bbl_sort(char string[], int length) {
int i,j;
char temp;
for (i = 0; i < length - 1; i++) {
for (j = 0; j < length - i - 1; j++) {
if (string[j] > string[j + 1]) {
// Swap the characters if they are in the wrong order
temp = string[j];
string[j] = string[j + 1];
string[j + 1] = temp;
} } } } int binary_srch_strng(char string[], char key, int left, int right) {
if (left > right) {
return -1; // Character not found in the string
} int mid = left + (right - left) / 2;
if (string[mid] == key) {
return mid; // Found the character at position 'mid'
} else if (string[mid] < key) {
return binary_srch_strng(string, key, mid + 1, right);
} else {
return binary_srch_strng(string, key, left, mid - 1);
} } int main() {
