int search(int ary[], int size, int key)
{ int mid,loc=-1,lb=0,up=size-1;
while (lb { mid=(lb+up)/2;
if (ary[mid]==key)
{ loc=mid;
break;
} else if (key { up=mid-1;
} else lb=mid+1;
} return loc;
} int bubble(int ary[], int size)
{ int i,j,temp;
for ( i = size-1; i >0; i--)
{ for ( j= 0; j <= i; j++)
{ if (ary[j]>ary[j+1])
{ temp=ary[j];
ary[j]=ary[j+1];
ary[j+1]=temp;
} } } return 0;
